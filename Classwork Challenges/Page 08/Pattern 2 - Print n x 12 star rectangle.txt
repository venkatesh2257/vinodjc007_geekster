import java.io.*;
import java.util.*;

public class Solution {

    public static void main(String[] args) {
        
        Scanner sc = new Scanner(System.in);
        int n = sc.nextInt();
        
        for(int j=1;j<=12;j++){
            //Pattern 1
            for(int i=1;i<=n;i++){
              System.out.print("*");
            }
            System.out.println();
        }
        
    }
}
